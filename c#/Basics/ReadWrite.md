# Write to console

    This is how you read/write to the console
    
# Code:

    using System;
    
    namespace <namespace name>
    {

        class <class name>
        {
            static void Main(string[] args)
            {
               Console.Write("Writing this text")
               Console.Read();
            }
        }
    }
    
    
    
    or
    
    
    namespace <namespace name>
    {

        class <class name>
        {
            static void Main(string[] args)
            {
               System.Console.Write("Writing this text")
               System.Console.Read();
            }
        }
    }
