## Code:


    int var = 211;

    switch (var)
    {
        case 15:
            Console.WriteLine("This is case 1");
            break;
        case 25:
            Console.WriteLine("This is case 2");
            break;
        default:
            Console.WriteLine("No other cases were met");
            break;
    }
